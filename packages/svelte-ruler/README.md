## üöÄ How to use
```html
<script>
import Ruler from "svelte-ruler";
import { onMount } from "svelte";

let ruler;
window.addEventListener("resize", () => {
    ruler.resize();
});
onMount(() => {
    requestAnimationFrame(() => {
        ruler.resize();
    });
})
</script>
<div class="ruler">
    <Ruler type="horizontal" bind:this={ruler}/>
</div>

```
